juego Juego_A :
	var fact[1] : int, i[1] : int;

function int factorial ( int x ) {
    if (x < 0){
        return false;
    }
    else{
        if (x == 0){
            
            fact = 1;
        }
        else {

            for i = 1 to x do {
            
                fact = fact * i;
            };
        return fact;

        };
    };
}


main (){
    
    fact = 1;
    output( factorial(5) );
}
